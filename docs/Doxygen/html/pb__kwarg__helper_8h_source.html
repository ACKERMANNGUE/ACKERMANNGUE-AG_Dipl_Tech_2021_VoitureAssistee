34 // Parse given positional and keyword arguments against a list of allowed arguments
35 // First n_ignore arguments are required arguments for which no keyword can be given.
36 #define PB_PARSE_ARGS(parsed_args, n_args, pos_args, kw_args, allowed_args, n_ignore) \
37  mp_arg_val_t parsed_args[MP_ARRAY_SIZE(allowed_args)]; \
38  mp_arg_parse_all(n_args - n_ignore, pos_args + n_ignore, kw_args, MP_ARRAY_SIZE(allowed_args), allowed_args, parsed_args)

43 // Generate table entry from an argument name, its requirements, and its default value

53 // Create the arguments table, parse it, and declare mp_obj_t's for each one
54 #define PB_PARSE_GENERIC(n_args, pos_args, kw_args, n_ignore, ...) STATIC const mp_arg_t allowed_args[] = { \
55  FOR_EACH_IDX(PB_ARG_DO, __VA_ARGS__) \
56 }; \
57  PB_PARSE_ARGS(parsed_args, n_args, pos_args, kw_args, allowed_args, n_ignore); \
58  FOR_EACH_IDX(GEN_ARG_OBJ, __VA_ARGS__)
