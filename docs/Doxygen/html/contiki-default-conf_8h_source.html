207 /* UIP_CONF_ND6_SEND_NS enables standard IPv6 Neighbor Discovery Protocol.
208  We enable it by default when IPv6 is used without RPL.
209  With RPL, the neighbor cache (link-local IPv6 <-> MAC address mapping)
210  is fed whenever receiving DIO and DAO messages. This is always sufficient
211  for RPL routing, i.e. to send to the preferred parent or any child.
212  Link-local unicast to other neighbors may, however, not be possible if
213  we never receive any DIO from them. This may happen if the link from the
214  neighbor to us is weak, if DIO transmissions are suppressed (Trickle
215  timer) or if the neighbor chooses not to transmit DIOs because it is
216  a leaf node or for any reason. */
217 #ifndef UIP_CONF_ND6_SEND_NS
218 #define UIP_CONF_ND6_SEND_NS (NETSTACK_CONF_WITH_IPV6 && !UIP_CONF_IPV6_RPL)
219 #endif /* UIP_CONF_ND6_SEND_NS */
220 /* UIP_CONF_ND6_SEND_NA allows to still comply with NDP even if the host does
221  not perform NUD or DAD processes. By default it is activated so the host
222  can still communicate with a full NDP peer. */
223 #ifndef UIP_CONF_ND6_SEND_NA
224 #define UIP_CONF_ND6_SEND_NA (NETSTACK_CONF_WITH_IPV6)
225 #endif /* UIP_CONF_ND6_SEND_NS */
